
export const SchemaParamFields = ({schema}) => {
  const columns = schema.models[0].columns;
  const description = schema.models[0].description;

  return (
    <>
    <p>{description.trim()}</p>
    <table style={{ width: '100%', borderCollapse: 'collapse', fontSize: '16px' }}>
      <thead>
        <tr style={{ background: '#4F46E5', color: 'white' }}>
          <th style={{ padding: '8px', border: '1px solid #ddd', textAlign: 'left' }}>Field</th>
          <th style={{ padding: '8px', border: '1px solid #ddd', textAlign: 'left' }}>Description</th>
        </tr>
      </thead>
      <tbody>
        {columns.map(col => (
          <tr key={col.name}>
            <td style={{ color: '#7C3AED', fontWeight: 'bold', padding: '8px', border: '1px solid #ddd', fontFamily: 'monospace' }}>
              {col.name}
            </td>
            <td style={{ padding: '8px', border: '1px solid #ddd' }}>
              <strong>{col.name.replace(/([A-Z])/g, ' $1').replace(/^./, str => str.toUpperCase())}:</strong> {col.description.trim()}
            </td>
          </tr>
        ))}
      </tbody>
    </table>
    </>
  );
};
